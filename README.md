# Console Accordion

## Sample input
```
const data: AccordionData = [
  {
    title: "Title 1",
    content: "Content 1",
  },
  {
    title: "Title 2",
    replace: "Title 2 - extended title with a very long text",
    content: "Content 2",
  },
  {
    title: "Title 3",
    content: "Content 3\nmore content",
  }
];

showAccordion(data);
```
